	{
	    int len;

	    cout << "Enter a number: ";
	    cin >> len;

	    for (int i = 0; i < len; i++)
	    {
		int j = i+1;
		while(j < len)
		{
			cout << “ “;
			j++;
		}
		cout << "#" << endl;
